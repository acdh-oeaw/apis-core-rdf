    def render(self, data, accepted_media_type=None, renderer_context=None):
        g = Graph()
        for result in data.get("results", []):
            match result:
                case tuple(_, _, _):
                    g.add(result)
                case other:
                    logger.debug("Could not add %s to RDF graph: not a tuple", other)
        return g.serialize(format=self.media_type)
