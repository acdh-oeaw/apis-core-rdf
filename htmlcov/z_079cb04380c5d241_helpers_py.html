22def datadump_get_queryset(additional_app_labels: list = []): 

23 """ 

24 This method is loosely based on the `dumpdata` admin command. 

25 It iterates throug the relevant app models and exports them using 

26 a serializer and natural foreign keys. 

27 Data exported this way can be reimported into a newly created Django APIS app 

28 """ 

29 

30 # get all APIS apps and all APIS models 

31 apis_app_labels = ["apis_relations", "apis_metainfo"] 

32 apis_app_models = [ 

33 model for model in apps.get_models() if model._meta.app_label in apis_app_labels 

34 ] 

35 

36 # create a list of app labels we want to iterate 

37 # this allows to extend the apps via the ?app_labels= parameter 

38 app_labels = set(apis_app_labels) 

39 app_labels |= set(additional_app_labels) 

40 

41 # look for models that inherit from APIS models and add their 

42 # app label to app_labels 

43 for model in apps.get_models(): 

44 if any(map(lambda x: issubclass(model, x), apis_app_models)): 

45 app_labels.add(model._meta.app_label) 

46 

47 # now go through all app labels 

48 app_list = {} 

49 for app_label in app_labels: 

50 app_config = apps.get_app_config(app_label) 

51 app_list[app_config] = None 

52 

53 models = serializers.sort_dependencies(app_list.items(), allow_cycles=True) 

54 

55 yield from datadump_get_objects(models)

87def get_html_diff(a, b, show_a=True, show_b=True, shorten=0): 

88 """ 

89 Create an colorized html represenation of the difference of two values a and b 

90 If `show_a` is True, colorize deletions in `a` 

91 If `show_b` is True, colorize insertions in `b` 

92 The value of `shorten` defines if long parts of strings that contains no change should be shortened 

93 """ 

94 

95 def style_remove(text): 

96 return f"<span class='diff-remove'>{text}</span>" 

97 

98 def style_insert(text): 

99 return f"<span class='diff-insert'>{text}</span>" 

100 

101 nones = ["", None] 

102 if a in nones and b in nones: 

103 result = "" 

104 elif a in nones: 

105 result = style_insert(b) if show_b else "" 

106 elif b in nones: 

107 result = style_remove(a) if show_a else "" 

108 else: 

109 result = "" 

110 a = str(a) 

111 b = str(b) 

112 codes = difflib.SequenceMatcher(None, a, b).get_opcodes() 

113 for opcode, a_start, a_end, b_start, b_end in codes: 

114 match opcode: 

115 case "equal": 

116 equal = a[a_start:a_end] 

117 if shorten and len(equal) > shorten: 

118 equal = equal[:5] + " ... " + equal[-10:] 

119 result += equal 

120 case "delete": 

121 if show_a: 

122 result += style_remove(a[a_start:a_end]) 

123 case "insert": 

124 if show_b: 

125 result += style_insert(b[b_start:b_end]) 

126 case "replace": 

127 if show_b: 

128 result += style_insert(b[b_start:b_end]) 

129 if show_a: 

130 result += style_remove(a[a_start:a_end]) 

131 return result

134def construct_lookup(value: str) -> tuple[str, str]: 

135 """ 

136 Helper method to parse input values and construct field lookups 

137 (https://docs.djangoproject.com/en/4.2/ref/models/querysets/#field-lookups) 

138 Parses user input for wildcards and returns a tuple containing the 

139 interpreted django lookup string and the trimmed value 

140 E.g. 

141 

142 - ``example`` -> ``('__icontains', 'example')`` 

143 - ``*example*`` -> ``('__icontains', 'example')`` 

144 - ``*example`` -> ``('__iendswith', 'example')`` 

145 - ``example*``-> ``('__istartswith', 'example')`` 

146 - ``"example"`` -> ``('__iexact', 'example')`` 

147 

148 :param str value: text to be parsed for ``*`` 

149 :return: a tuple containing the lookup type and the value without modifiers 

150 """ 

151 

152 if value.startswith("*") and not value.endswith("*"): 

153 value = value[1:] 

154 return "__iendswith", value 

155 

156 elif not value.startswith("*") and value.endswith("*"): 

157 value = value[:-1] 

158 return "__istartswith", value 

159 

160 elif value.startswith('"') and value.endswith('"'): 

161 value = value[1:-1] 

162 return "__iexact", value 

163 

164 else: 

165 if value.startswith("*") and value.endswith("*"): 

166 value = value[1:-1] 

167 return "__icontains", value
